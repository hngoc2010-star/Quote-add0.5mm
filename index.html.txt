<script>

function format(x){
if(x==""||x==null) return ""
return Number(x).toLocaleString()
}

document.querySelectorAll("input").forEach(i=>{
i.addEventListener("input",calc)
})

function val(id){
let v=document.getElementById(id).value.replace(/,/g,'')
return Number(v)||0
}

function calc(){

let thickness=val("thickness")
let bundles=val("bundles")
let pcsBundle=val("pcsBundle")

let exwPcs=val("exwPcs")
let exwCbm=val("exwCbm")

let rate=val("rate")
let logistics=val("logistics")
let cfr=val("cfr")

let cbmBuy=val("cbmBuy")
let cbmSell=val("cbmSell")

let thSell=thickness+0.5
document.getElementById("thSell").innerText=thSell

let pcs=bundles*pcsBundle
document.getElementById("pcs").innerText=format(pcs)

let cbmDiff=cbmSell-cbmBuy
document.getElementById("cbmDiff").innerText=format(cbmDiff)

let cost=0

if(exwPcs>0)
cost=exwPcs*pcs*rate
else
cost=exwCbm*cbmBuy*rate

cost+=logistics

let sell=cfr

let profit=sell-cost

let profitCbm=0
if(cbmSell>0) profitCbm=profit/cbmSell

let profitTotal=document.getElementById("profitTotal")
let profitCbmBox=document.getElementById("profitCbm")

profitTotal.innerText=format(Math.round(profit))
profitCbmBox.innerText=format(Math.round(profitCbm))

if(profit>=0){
profitTotal.className="result blue"
profitCbmBox.className="result blue"
}
else{
profitTotal.className="result red"
profitCbmBox.className="result red"
}

document.querySelectorAll("input").forEach(i=>{
let v=i.value.replace(/,/g,'')
if(v!="") i.value=format(v)
})

}
